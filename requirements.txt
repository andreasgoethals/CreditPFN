# =============================================================================
# CreditPFN — runtime dependencies

# Used by every script in src/data/ and (later) src/train, src/eval, src/model.
# Pinned to conservative major-version ranges. We version-pin minor where the
# upstream API has changed in a breaking way recently.
#
# Python version: ** USE 3.11 OR 3.12 ** (do NOT use 3.14)
# -----------------------------------------------------------------------------
# scikit-learn, torch, and several other compiled packages do not yet ship
# prebuilt wheels for Python 3.14 (cp314). Pip falls back to compiling from
# source via MSVC, which crashes mid-build on Windows. Python 3.12 is the
# sweet spot: every dep below has a cp312 wheel on PyPI as of late 2025.
#
# To switch: install Python 3.12 from python.org, then
#     py -3.12 -m venv .venv --prompt CreditPFN
#     .venv/Scripts/activate
#     python -m pip install --upgrade pip
#     pip install -r requirements.txt
#
# Refresh with:  pip install -r requirements.txt
# =============================================================================

# --- Core deep-learning stack ------------------------------------------------
torch>=2.2,<3.0

# --- TabPFN (sklearn API + finetuning wrappers + internal preprocessing) ------
# Note on versioning: this project's `src/train/` code is written against
# the **canonical TabPFN API documented in repositories/TabPFN .txt** —
# specifically:
#   * load_model_criterion_config returns 4-tuple (list[Architecture],
#     criterion, list[ArchitectureConfig], InferenceConfig)
#   * version: Literal["v2", "v2.5", "v2.6", "v3"]
#   * download_if_not_exists keyword
#   * PerFeatureTransformer.forward(x, y, *, only_return_standard_out,
#     categorical_inds=list[list[int]], ...) — auto-derives
#     single_eval_pos from len(y)
# PyPI's `tabpfn` currently caps at 2.2.x with a different, narrower API
# (3-tuple return, version="v2" only, `download` keyword). The code will
# only run end-to-end against the newer release; on VSC install from a
# Prior Labs wheel / git source providing v2.5+ API, NOT from plain PyPI.
# Versions ≥6 on PyPI are an unrelated package that grabbed the name.
tabpfn>=2.2,<3.0
